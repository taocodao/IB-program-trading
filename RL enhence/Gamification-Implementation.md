  profit_factor DECIMAL(5,2),
  journal_entry TEXT,
  created_at TIMESTAMP
);

-- Leaderboard (materialized view)
CREATE MATERIALIZED VIEW user_leaderboard AS
SELECT 
  user_id,
  (win_rate * 0.40 + 
   risk_management * 0.30 + 
   consistency * 0.20 + 
   learning * 0.10) as discipline_score,
  ROW_NUMBER() OVER (ORDER BY discipline_score DESC) as rank
FROM (
  SELECT
    t.user_id,
    AVG(CASE WHEN t.profit_loss > 0 THEN 1 ELSE 0 END) * 100 as win_rate,
    COUNT(CASE WHEN t.has_stop_loss THEN 1 END)::float / COUNT(*) * 100 as risk_management,
    -- consistency calculation
    -- learning calculation
  FROM trades t
  WHERE t.created_at > NOW() - INTERVAL '30 days'
  AND t.user_id NOT IN (select user_id from suspicious_accounts)
  GROUP BY t.user_id
);

-- Challenges
CREATE TABLE challenges (
  id UUID PRIMARY KEY,
  title VARCHAR(200),
  description TEXT,
  type VARCHAR(20), -- 'weekly', 'monthly', 'group'
  start_date DATE,
  end_date DATE,
  difficulty VARCHAR(10), -- 'easy', 'medium', 'hard'
  criteria JSONB, -- how to complete
  reward_xp INT,
  reward_badge_id UUID,
  participants_count INT
);

CREATE TABLE user_challenges (
  id UUID PRIMARY KEY,
  user_id UUID,
  challenge_id UUID,
  progress FLOAT,
  completed BOOLEAN,
  completed_at TIMESTAMP
);
